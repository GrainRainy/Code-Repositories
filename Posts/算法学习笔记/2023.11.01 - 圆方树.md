# 圆方树


## 构建 $\rm Build$

```cpp
void tarjan(int u) {
	dfn[u] = low[u] = ++ stp, stk[++ top] = u, cnt ++;
	for (int i = head[u], v; ~i; i = edge[i].nxt) {
		if (!dfn[v = edge[i].to]) {
			tarjan(v, u);
			low[u] = min(low[u], low[v]);
			if (low[v] >= dfn[u]) {
				w[++ dcc] = 1; int tmp;
				g[dcc].emplace_back(u);
				g[u].emplace_back(dcc);
				do {
					tmp = stk[top --], w[dcc] ++;
					g[dcc].emplace_back(tmp);
					g[tmp].emplace_back(dcc);
				} while (tmp != v);
			}
		} else low[u] = min(low[u], dfn[v]);
	}
}
```
