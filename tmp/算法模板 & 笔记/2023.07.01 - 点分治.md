## 静态点分治

#### 前置数组

```cpp
const int N = 1e4 + 10;
const int M = 1e2 + 10;
const int DIS = 1e8 + 10;
int n, m, u, v, w, k;
int root, sum, siz[N], mx[N], dis[N];
int head[N], idx;
struct Node { int nxt, to, w; }edge[N << 1];
bool del[N], save[DIS];
int query[M], rem[DIS];
bool test[M];
int q[N], top; // 待清空队列  
```

#### 寻找子树重心

```cpp
void getRt(int u, int fa) {
	siz[u] = 1, mx[u] = 0;
	for (int i = head[u], j; ~i; i = edge[i].nxt) {
		j = edge[i].to;
		if (j == fa or del[j]) continue;
		getRt(j, u);
		siz[u] += siz[j];
		mx[u] = max(mx[u], siz[j]);
	}
	mx[u] = max(mx[u], sum - siz[u]);
	if (mx[u] < mx[root]) root = u;
}
```

#### 更新子树到根距离

```cpp
void getDis(int u, int fa) {
	rem[++ rem[0]] = dis[u];
	for (int i = head[u], j; ~i; i = edge[i].nxt) {
		j = edge[i].to;
		if (j == fa or del[j]) continue;
		dis[j] = dis[u] + edge[i].w;
		getDis(j, u);
	}
}
```

#### 拼合经过根的路径

```cpp
void calc(int u) {
	top = 0;
	for (int i = head[u], j; ~i; i = edge[i].nxt) {
		j = edge[i].to;
		if (del[j]) continue;
		dis[j] = edge[i].w;
		getDis(j, u);
		
		for (int j = rem[0]; j; -- j)
			for (int k = 1; k <= m; ++ k)
				if (query[k] >= rem[j]) 
					test[k] |= save[query[k] - rem[j]];
		
		for (int &j = rem[0]; j; -- j)
			q[++ top] = rem[j], save[rem[j]] = 1;
	}
	for (int i = 1; i <= top; ++ i)
		save[q[i]] = 0;
}
```

#### 分治子树

```cpp
void solve(int u) {
	del[u] = save[0] = true;
	calc(u);
	for (int i = head[u], j; ~i; i = edge[i].nxt) {
		j = edge[i].to;
		if (del[j]) continue;
		sum = siz[j];
		mx[root = 0] = DIS;
		getRt(j, 0);
		solve(root);
	}
}
```

for (int i = 1; i <= m; ++ i) input(query[i]);
